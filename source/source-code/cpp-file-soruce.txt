void trigometer_by_angle(){
    double angle, deg;

    system("cls");
    system("color 0c");
    int a;
    do {
        cout<<"0) sin \n1) cos \n2) tan \n3) csc \n4) sec \n5) cot \n\n";
        cin>>a;
    } while (a<0 || a>5);

    system("cls");
    cout<<"Enter the angle in degrees: ";
    cin>>deg;
    angle = deg * M_PI / 180.0;

    switch (a)
    {
    case 0:
        cout<<"\n\n Answer is sin("<<deg<<") = "<<sin(angle)<<endl;
        break;
    case 1:
        cout<<"\n\n Answer is cos("<<deg<<") = "<<cos(angle)<<endl;
        break;   
    case 2:
        if (fmod(deg,180) != 90){
            cout<<"\n\n Answer is tan("<<deg<<") = "<<tan(angle)<<endl;
        } else {
            cout<<"Undefined! tan("<<deg<<")"<<endl;
        }
        break;
    case 3:
        if (sin(angle) != 0){
            cout<<"\n\n Answer is csc("<<deg<<") = "<<1/sin(angle)<<endl;
        } else {
            cout<<"Undefined! csc("<<deg<<")"<<endl;
        }
        break;   
    case 4:
        if (cos(angle) != 0){
            cout<<"\n\n Answer is sec("<<deg<<") = "<<1/cos(angle)<<endl;
        } else {
            cout<<"Undefined! sec("<<deg<<")"<<endl;
        }
        break;
    case 5:
        if (tan(angle) != 0){
            cout<<"\n\n Answer is cot("<<deg<<") = "<<1/tan(angle)<<endl;
        } else {
            cout<<"Undefined! cot("<<deg<<")"<<endl;
        }
        break;      
    }
    cout<<endl;
    system("pause");
    show_main_menu();
}
